create new label for the hours in which bikes were rented

In [4]:
def assign_label(hour):    
    if((hour>=6) & (hour<12)):
        return 1
    elif((hour>=12) & (hour<18)):
        return 2
    elif((hour>=18) & (hour<24)):
        return 3
    elif((hour>=0) & (hour<6)):
        return 4
    
bike_rentals['time_label'] = bike_rentals['hr'].apply(lambda x: assign_label(x))
